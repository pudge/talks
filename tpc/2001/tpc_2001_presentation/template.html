[% stories = Slash.db.getNewStories(form.section) %]
[% INCLUDE titlebar
    title => Slash.db.getSection(form.section).title %]

    });
    "</B> (";
        Slash.timeCalc("$story.5 12:00", "%Y-%m-%d");

END %]
