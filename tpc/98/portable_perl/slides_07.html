  • use File::Spec;

    • Comes with 5.004_05 and 5.005.
    • Does not work with all platforms (yet)
      use File::Spec;
  chdir(File::Spec->updir());
  $file = File::Spec->catfile(
    File::Spec->curdir(), 'temp', 'file.txt');
  # $file is now './temp/file.txt' on Windows and Unix,
  # and ':temp:file.txt' on Mac OS
